## Quick Example

<CodeGroup>
```javascript Node.js
import WebSocket from 'ws';

const ws = new WebSocket('wss://api.tik.tools?uniqueId=streamer_name&apiKey=YOUR_KEY');

ws.on('message', (data) => {
  const event = JSON.parse(data);
  if (event.event === 'chat') {
    console.log(`${event.data.user.uniqueId}: ${event.data.comment}`);
  }
});
```

```python Python
import asyncio, websockets, json

async def listen():
    url = "wss://api.tik.tools?uniqueId=streamer_name&apiKey=YOUR_KEY"
    async with websockets.connect(url) as ws:
        async for message in ws:
            event = json.loads(message)
            if event["event"] == "chat":
                print(f"{event['data']['user']['uniqueId']}: {event['data']['comment']}")

asyncio.run(listen())
```

```bash cURL
# Test with wscat:
npx wscat -c "wss://api.tik.tools?uniqueId=streamer_name&apiKey=YOUR_KEY"

# Or check if a user is live via REST:
curl "https://api.tik.tools/webcast/check_alive?apiKey=YOUR_KEY&unique_id=streamer_name"
```
